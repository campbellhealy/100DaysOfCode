'''
    Like the previous hurdels game , but this time the number, height and spacing 
    of hurdles changes with each game.

    https://reeborg.ca/reeborg.html
'''
def turn_right():
    turn_left()
    turn_left()
    turn_left()


def hop():
    turn_left()
    while wall_on_right():
        move()
    turn_right()
    move()
    turn_right()
    while front_is_clear():
        move()
    turn_left()


while not at_goal():
    if wall_in_front():
        hop()
    else:
        move()
